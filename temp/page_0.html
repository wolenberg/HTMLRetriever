I'm wondering why I always meet "Unable to locate the specified class: Session.php" after i click the login button.

What I'm trying to do is I have a controller for login. And after the login validate is true, I want to redirect it to another controller.

The action for login view button is in Controller_A Controller_A

public function login()
        {
            $postData   = $this->input->post();
            $validate   = $this->model_Auth->validate_login($postData);
            if ($validate)
            {
                
                $newdata = array(
                    'MEMBER_ID'             => $validate[0]->MEMBER_ID,
                    'MEMBER_NAME'           => $validate[0]->MEMBER_NAME,
                    'MEMBER_CARD_NO'        => $validate[0]->MEMBER_CARD_NO,
                    'MEMBER_PLACE_OF_BIRTH' => $validate[0]->MEMBER_PLACE_OF_BIRTH,
                    'MEMBER_DATE_OF_BIRTH'  => $validate[0]->MEMBER_DATE_OF_BIRTH,
                    'MEMBER_ADDRESS'        => $validate[0]->MEMBER_ADDRESS,
                    'MEMBER_POIN'           => $validate[0]->MEMBER_POIN,
                    'MEMBER_KUPON'          => $validate[0]->MEMBER_KUPON,
                    //'password'            => $validate[0]->password,
                    'logged_in'             => TRUE,   
                );
                $this->session->set_userdata($newdata);
                redirect(base_url("Controller_B"));

Controller_B

class Controller_B extends CI_Controller 
    {   
        public function __Construct() 
        {
            parent::__Construct();
            $this->load->model("model_Member");
            date_default_timezone_set('Asia/Jakarta');
        }
        public function index() 
        {
            if($this->session->userdata('logged_in'))

            }
            else 
            {
                $data = array('alert' => false);
                $this->load->view('home/login', $data);
            }
        }
